Display Fibonacci Series using Recursion

 #include< stdio.h >
#include< conio.h >
main()
{
   int n,i,t;
   printf("\n Enter the number of terms to be generated");
   scanf("%d",&n);
   for(i=1;i<=n;i++)
   {
      t=fib(i);
      printf("\t%d",t);
   }
   getch();
}
fib(int n)
{
   if(n==1)
       return(0);
   else if(n==2)
       return(1);
   else
       return(fib(n-1)+fib(n-2));
}

Factorial of a number using Recursion

 #include< stdio.h >
#include< conio.h >
int fact(int);
int main()
{
     int num,f;
     printf("\n Enter the number");
     scanf("%d",&num);
     f=fact(num); 
     printf("\n The factorial of %d=%d",num,f);
     getch();
}
int fact(int n)
{
     if(n==0)
         return(1);
     else
         return(n*fact(n-1));
}

Funtion Call including Passing Arguments by Reference

Funtion Call including Passing Arguments by Reference

 #include < stdio.h >

void swap( int *p1, int *p2 );

int main()
{
   int a = 10;
   int b = 20;

   printf("Before: Value of a = %d and value of b = %d\n", a, b );
   swap( &a, &b );
   printf("After: Value of a = %d and value of b = %d\n", a, b );
}

void swap( int *p1, int *p2 )
{
    int t;

    t = *p2;
    *p2 = *p1;
    *p1 = t;
   printf("Value of a (p1) = %d and value of b(p2) = %d\n", *p1, *p2 );
}

Reverse of a number using Recursion

Reverse of a number using Recursion

 #include< stdio.h >
#include< conio.h >
int reverse(int);
main()
{
   int n,rev;
   printf("Enter an integer number");
   scanf("%d",&n);
   rev=reverse(n);
   printf("Reversed number is =%d",rev);
   getch();
}
int reverse(int n)
{
   int r,rev=0;
   while(n  > 0)
   {
      r=n%10;
      n=n/10;
      rev=rev*10+r;
   }
   return(rev);
}
