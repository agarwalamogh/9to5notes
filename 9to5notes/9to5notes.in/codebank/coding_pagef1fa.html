 int factorial(int x)
 {
    int f=1,i;
    for(i=x; i > = 1;i--)
     {
      f=f*i;
     }
    return(f);
 }

Program to Find Fibonacci Series

Program to Find Fibonacci Series

 	#include < stdio.h >
	#include < conio.h >
	main()
	{
		int n,x=0,y=1,z,i;
		printf("Enter the value of n:-");
		scanf("%d",&n);
		if(n<1)
			printf("Invalid Terms");
		if(n==1)
			printf("%d",x);
		if(n==2)
			printf("%d\t%d",x,y);
		if(n > 2)
		{
			printf("%d\t%d",x,y);
			for(i=3;i<=n;i++)
			{
				z=x+y;
				printf("\t%d",z);
				x=y;
				y=z;
			}
		}
		getch();
	}

Program using Return Statement

Program using Return Statement

 #include < stdio.h >
int sumDigits();
main()
{
    int sum = sumDigits();
    printf("%d\n", sum);
    return sum;
}

int sumDigits()
{
    int sum = 0;
    int digit;
    for(digit = 0; digit <= 9; ++digit)
    {
        sum += digit;
    }
    return sum;
}
