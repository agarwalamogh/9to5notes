#include< stdio.h >
#include< conio.h >
main()
{
   int n,i,t;
   printf("\n Enter the number of terms to be generated");
   scanf("%d",&n);
   for(i=1;i<=n;i++)
   {
      t=fib(i);
      printf("\t%d",t);
   }
   getch();
}
fib(int n)
{
   if(n==1)
       return(0);
   else if(n==2)
       return(1);
   else
       return(fib(n-1)+fib(n-2));
}
